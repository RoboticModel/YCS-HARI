AUTO-ROOM 
int fan = 8;
int light = 9;
int plug = 10;
char command;

void setup() {
  Serial.begin(9600);
  Serial.println("AutoRoom Initalized!");
  pinMode(fan, OUTPUT);
  pinMode(light, OUTPUT);
  pinMode(plug, OUTPUT);
}

void loop() {
  command = Serial.read();
  if (command != -1) { 
    if(command == 'L'){
      digitalWrite(light, HIGH);
    }else if(command == 'l'){
      digitalWrite(light, LOW);
    }else if(command == 'F'){
      digitalWrite(fan, HIGH);
    }else if(command == 'f'){
      digitalWrite(fan, LOW);
    }else if(command == 'P'){
      digitalWrite(plug, HIGH);
    }else if(command == 'p'){
      digitalWrite(plug, LOW);
    }
  } 

  delay(10); 
}
